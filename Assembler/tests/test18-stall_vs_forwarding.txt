@This example is taken from the HiP lecture example and is meant to showcase the difference between stall and forwarding

.data

.word 1

.text

loop:		@ stall										| forwarding
mov r3, #3	@ 5, 19										| 5, 13	
ldr r1, [r2]	@ 6, 20										| 6, 14
add r1, r1, #1	@ 10, 24									| 8, 16 (here one mandatory stall is used to get the value from the MA stage)
str r2, r1	@ 14 (written to operand memory on cycle 13, but left pipeline on 14), 28	| 9, 17
subs r4, r3, r1	@ 15, 29									| 10, 18
jne loop	@ 16, 30									| 11, 19
